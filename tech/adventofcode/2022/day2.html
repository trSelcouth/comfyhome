          #read input
          input = open("input2.txt", "r").readlines()
          
          PLAY_POINTS = {"A": 1, "B": 2, "C": 3}
          BEATS = {"A": "B", "B": "C", "C": "A"}
          LOSES_TO = {"A": "C", "B": "A", "C": "B"}
          
          LOSE_POINTS = 0
          DRAW_POINTS = 3
          WIN_POINTS = 6
          
          ROCK = "A"
          PAPER = "B"
          SCISSORS = "C"
          WIN = "Z"
          DRAW = "Y"
          LOSE = "X"
          
          # ROCK BEATS SCISSORS
          # SCISSORS BEAT PAPER
          # PAPER BEATS ROCK
          
          points = 0
          
          for game in input:
              RESULT = game.split(' ')[1].strip()
              OPPONENT = game.split(' ')[0].strip()
          
              if RESULT == DRAW:
                  points += DRAW_POINTS + PLAY_POINTS[OPPONENT]
              elif RESULT == WIN:
                  points += WIN_POINTS + PLAY_POINTS[BEATS[OPPONENT]]
              elif RESULT == LOSE:
                  points += LOSE_POINTS + PLAY_POINTS[LOSES_TO[OPPONENT]]
          
          print("TOTAL POINTS:", str(points))
