        input = open("input1.txt", "r").readlines()
        elves = []
        calories = 0
        best_three = 0
        
        #for each line
        for line in input:
            #if it's empty - new elf. Save calories of previous elf and reset counter
            if len(line) <= 1:
                elves.append(calories)
                calories = 0
            #otherwise - add up calories to counter
            else:
                calories += int(line)
        
        #find elf with most calories, add up to counter, remove from elf list and repeat three times
        for i in range(1,4):
            max_elf = max(elves)
            best_three += max_elf
            elves.remove(max_elf)
            print("Maximum calories on one elf: " + str(max_elf))
            
        print("Calories on three maxed elves: " + str(best_three))
